romanian_conference.0 = {
	type = country_event
	placement = root

	title = romanian_conference.0.t
	desc = romanian_conference.0.d # probably want a triggered desc here
	flavor = romanian_conference.0.f

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 3

	immediate = {
	}
	option = {
		name = romanian_conference.0.a
		default_option = yes
		change_global_variable = {
			name = unite_the_principalities
			add = 1
		}
	}
	option = {
		name = romanian_conference.0.b
		change_global_variable = {
			name = unite_the_principalities
			add = -1
		}
	}
}

		}

			random_scope_state = {
			limit = {

			}
		}
	}
	option = {
		name = romanian_conference.11.b
		change_infamy = 25
	}
}
