namespace = treaty_of_paris_events

treaty_of_paris_events.1 = {
	type = country_event
	placement = ROOT

	title = treaty_of_paris_events.1.t
	desc = treaty_of_paris_events.1.d
	flavor = treaty_of_paris_events.1.f

	cooldown = { days = short_modifier_time }

	event_image = {
		video = "southamerica_aristocrats"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
			c:TUR ?= this 
			NOT = {
				has_variable = treaty_of_paris_initiation_var 
			}
			is_diplomatic_play_enemy_of = c:RUS
	}

	immediate = {
		set_variable = treaty_of_paris_initiation_var
		c:RUS ?= {
			save_scope_as = russia_scope
		}
	}

	option = {
		name = treaty_of_paris_events.1.a
		default_option = yes
		ai_chance = {
			base = 95
		}
		c:RUS ?= {
			trigger_event = { id = treaty_of_paris_events.2 popup = yes }
		}
		custom_tooltip = if_they_accept_tt
		custom_tooltip = treaty_of_paris_explanation_tt
		show_as_tooltip = {
			add_journal_entry = {
				type = je_congress_of_paris
			}
			c:RUS ?= {
				add_journal_entry = {
					type = je_congress_of_paris
				}
			}
			every_diplomatic_play = {
				limit = {
					initiator_is = c:RUS
					target_is = c:TUR 
				}
				end_play = yes
			}
			every_country_in_europe = {
				limit = {
					country_rank >= rank_value:great_power
					NOR = {
						root ?= this 
						c:RUS ?= this
						c:TUR ?= this
					}
				}
				trigger_event = { id = treaty_of_paris_events.3 popup = yes }
			}
		}
	}


	option = {
		name = treaty_of_paris_events.1.b
		ai_chance = {
			base = 5
		}
		c:RUS ?= {
			change_relations = { country = root value = -80 }
			create_diplomatic_catalyst = {
				type = catalyst_event_negative
				target = root
			}
		}
		every_country_in_europe = {
			limit = {
				country_rank >= rank_value:great_power
				NOT = { this = root }
			}
			change_relations = { country = root value = -40 }
		}
		add_radicals = {
			value = 0.05
		}
	}
}

treaty_of_paris_events.4 = {
	type = country_event
	placement = ROOT

	title = treaty_of_paris_events.4.t
	desc = treaty_of_paris_events.4.d
	flavor = treaty_of_paris_events.4.f

	cooldown = { days = normal_modifier_time }

	event_image = {
		video = "ep1_transfer_of_authority"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		c:RUS ?= {
			save_scope_as = russia_scope
		}
		c:TUR ?= {
			save_scope_as = ottoman_scope
		}
	}

	option = {
		name = treaty_of_paris_events.4.a
		default_option = yes
		ai_chance = {
			base = 95
		}
		if = {
			limit = {
				c:TUR ?= this
			}
			custom_tooltip = {
				text = treaty_of_paris_outcome_tt
				enforce_treaty_of_paris = yes
			}
		}
		else = {
			show_as_tooltip = {
				custom_tooltip = {
					text = treaty_of_paris_outcome_tt
					enforce_treaty_of_paris = yes
				}
			}
		}
		if = {
			limit = {
				NOR = {
					c:RUS ?= this
					c:TUR ?= this
				}
				NOT = {
					any_scope_treaty = {
						binds = c:RUS
						any_scope_article = {
							has_type = guarantee_independence
						}
					}
				}
			}
			create_treaty = {
				first_country = root
				second_country = c:TUR

				is_draft = no
				binding_period = { years = 5 }

				articles_to_create = {
					{
						article = guarantee_independence
						source_country = root
						target_country = c:TUR
					}
				}
			}
		}
		if = {
			limit = {
				NOR = {
					c:RUS ?= this
					c:TUR ?= this
				}
			}
			add_modifier = {
				name = guarantors_of_peace_modifier
				days = normal_modifier_time
			}
		}
}   }
