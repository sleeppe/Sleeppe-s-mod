polish_constitution.4 = {
	type = country_event
	placement = root

	title = polish_constitution.4.t
	desc = polish_constitution.4.d # probably want a triggered desc here
	flavor = polish_constitution.4.f

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 3

	immediate = {
	}
	option = {
		name = polish_constitution.4.a
		c:POL = {
			annex_with_incorporation = c:KRA
		}
		c:AUS = {
			add_loyalists = {
				culture = cu:polish
				value = 0.15
			}
		}
		c:POL = {
			increase_autonomy = yes
		}
		if = {
			limit = {
				has_global_variable = polish_constitution_global
			}
			change_global_variable = {
				name = polish_constitution_global
				add = 1
			}
		}
	}
	option = {
		name = polish_constitution.4.b
		c:AUS = {
			add_radicals = {
				culture = cu:polish
				value = 0.20
			}
		}
		c:POL = {
			change_relations = {
				country = c:AUS
				value = -30
			}
		}
		c:POL = {
			trigger_event = polish_constitution.5
		}
	}
}
