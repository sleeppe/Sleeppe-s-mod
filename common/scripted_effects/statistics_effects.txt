#Require input: $name$, $value
stat_global_list_effect = {
	clear_global_variable_list = stat_list_$name$

	set_global_variable = {
		name = stat_list_count_$name$
		value = 0
	}
	ordered_country = {
		limit = {
			is_player = yes
			statistics_is_valid = yes
			$value$ > 0
		}
		order_by = {
			value = $value$
			multiply = global_var:stat_direction
		}
		max = 20 # TODO: Make configurable
		check_range_bounds = no

		add_to_global_variable_list = {
			name = stat_list_$name$
			target = this
		}
		set_variable = {
			name = stat_value_$name$
			value = $value$
		}
		change_global_variable = {
			name = stat_list_count_$name$
			add = 1
		}
	}
	ordered_country = {
		limit = {
			is_player = no
			statistics_is_valid = yes
			$value$ > 0
		}
		order_by = {
			value = $value$
			multiply = global_var:stat_direction
		}
		max = {
			value = 20 # TODO: Make configurable
			subtract = global_var:stat_list_count_$name$
		}
		check_range_bounds = no

		add_to_global_variable_list = {
			name = stat_list_$name$
			target = this
		}
		set_variable = {
			name = stat_value_$name$
			value = $value$
		}
	}
	ordered_in_global_list = {
		variable = stat_list_$name$
		order_by = $value$
		max = 1
		check_range_bounds = no
		set_global_variable = {
			name = stat_top_$name$
			value = $value$
		}
	}
	every_in_global_list = {
		variable = stat_list_$name$
		set_variable = {
			name = stat_var_$name$
			value = {
				value = $value$
				divide = global_var:stat_top_$name$
				multiply = 950 # TODO: Make configurable
				min = 50 # So that the text is always visible
			}
		}
	}

	# Suppress error
	if = {
		limit = {
			always = no
			any_country = {
				has_variable = stat_value_$name$
				has_variable = stat_var_$name$
			}
		}
	}
}

reverse_stat_direction = {
	if = {
		limit = {
			has_global_variable = stat_direction
		}
		change_global_variable = {
			name = stat_direction
			multiply = -1
		}
	} else = {
		set_global_variable = { name = stat_direction value = -1 }
	}
}
