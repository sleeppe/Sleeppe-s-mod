				}
			}
			c:TUR ?= {

					limit = {

			}

	}

		limit = {
			has_global_variable = favour_ottoman_global_var
		}

			limit = {
				exists = c:MOL
			}

					limit = {
						state_region = s:STATE_BESSARABIA
					}
					set_state_owner = c:MOL
				}
			}
		}
		else_if = {
			limit = {
				NOT = {
					exists = c:MOL
				}
			}
			c:RUS ?= {
				random_scope_state = {
					limit = {
						state_region = s:STATE_BESSARABIA
					}

			}

					}
				}

				random_scope_state = {
					limit = {
						state_region = s:STATE_KARS
					}
					set_state_owner = c:TUR
				}
			}
		}
		c:TUR ?= {
			take_on_scaled_debt = {
				who = c:RUS
				value = c:RUS.scaled_debt
			}
		}
	}
	else_if = {
		limit = {
			has_global_variable = partition_compromise_global_var
		}

			set_owner_of_provinces = {

				}

					create_diplomatic_pact = {
						country = prev
						type = personal_union
					}
				}
			}
		}
	}
	else_if = {
		limit = {
			has_global_variable = luxembourg_compromise_global_var
		}
		if = {
			limit = {
				exists = c:LUX
			}
			s:STATE_WALLONIA = {
				set_owner_of_provinces = {
					country = c:LUX

						x5180a0 x3a5fb9
					}
				}
			}
		}
		else = {
			s:STATE_WALLONIA = {
				set_owner_of_provinces = {
					country = c:TUR
					provinces = {
						x5180a0 x3a5fb9
					}
				}
			}
		}
	}
}
