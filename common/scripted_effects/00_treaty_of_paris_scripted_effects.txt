enforce_treaty_of_paris = {
	if = {
		limit = {
			has_global_variable = favour_russia_global_var
		}

		if = {
			c:TUR ?= {
				random_scope_state = {
					limit = {
						state_region = s:STATE_DOBRUDJA
					}
					set_state_owner = c:RUS
				}
			}
			c:TUR ?= {
					limit = {
						state_region = s:STATE_KARS
					}
					set_state_owner = c:RUS 
				}
			}
		}
	else_if = {
		limit = {
			has_global_variable = favour_ottoman_global_var
		}
		if = {
			limit = {
				exists = c:MOL
			}
			c:RUS ?= {
				random_scope_state = {
					limit = {
						state_region = s:STATE_BESSARABIA
					}
					set_state_owner = c:MOL
				}
			}
			scope:ottoman_scope = {
				transfer_subject = c:MOL
			}
			scope:ottoman_scope = {
				transfer_subject = c:WAL
			}
	}
		else_if = {
			limit = {
				NOT = {
					exists = c:MOL
				}
			}
			c:RUS ?= {
				random_scope_state = {
					limit = {
						state_region = s:STATE_BESSARABIA
					}
					set_state_owner = c:TUR
				}
			}
		}
		s:STATE_DOBRUDJA = {
			set_owner_of_provinces = {
				country = c:TUR
				provinces = {
					x50F5CB
				}
			}
		}
		if = {
			limit = {
				c:RUS ?= {
					any_scope_state = {
						state_region = s:STATE_KARS
					}
				}
			}
			c:RUS ?= {
				random_scope_state = {
					limit = {
						state_region = s:STATE_KARS
					}
					set_state_owner = c:TUR
				}
			}
		}
	}
	else_if = {
		limit = {
			has_global_variable = the_romanian_solution_global_var
		}
			c:WAL ?= {
				annex = c:MOL
				change_tag = ROM
			}
			s:STATE_BESSARABIA = {
				set_owner_of_provinces = {
					country = c:ROM 
					provinces = {
						x712161 x6BF7B8 xB0E0A0
					}
				}
			}
		}
		s:STATE_DOBRUDJA = {
			set_owner_of_provinces = {
				country = c:TUR
				provinces = {
					x50F5CB
			}
		}
		scope:ottoman_scope = {
				transfer_subject = c:ROM
		}
	}	
}
