bpmparl_clear_party_cache = {
	set_variable = { name = bpm_seat_count_$KEY$ value = 0 }
	set_variable = { name = bpm_$COLOR$_parl_min value = 0 }
	set_variable = { name = bpm_$COLOR$_parl_max value = -1 }
	if = {
		limit = { has_variable = bpm_pointer_$KEY$_party }
		remove_variable = bpm_pointer_$KEY$_party
	}
}

bpmparl_cache_party_seats = {
	if = {
		limit = {
			any_active_party = {
				party_has_members = yes
				is_party_type = $PARTY_TYPE$
			}
		}
		random_active_party = {
			limit = {
				party_has_members = yes
				is_party_type = $PARTY_TYPE$
			}
			save_scope_as = bpmparl_target_party
		}
		set_variable = { name = bpm_pointer_$KEY$_party value = scope:bpmparl_target_party }
		set_variable = {
			name = bpm_seat_count_$KEY$
			value = {
				value = scope:bpmparl_target_party.bpmparl_party_total_clout
				divide = bpmparl_total_active_party_clout
				multiply = bpmparl_number_of_seats
				floor = yes
			}
		}
		change_variable = {
			name = bpmparl_total_counted_seats
			add = var:bpm_seat_count_$KEY$
		}
		if = {
			limit = {
				var:bpm_seat_count_$KEY$ > var:bpmparl_winning_vote_count
			}
			set_variable = {
				name = bpmparl_winning_vote_count
				value = var:bpm_seat_count_$KEY$
			}
			set_variable = {
				name = bpmparl_winning_party
				value = flag:$KEY$
			}
		}
	}
}

bpmparl_allocate_remaining_seats = {
	if = {
		limit = {
			has_variable = bpmparl_winning_party
			var:bpmparl_leftover_seats > 0
			var:bpmparl_winning_party = flag:$KEY$
		}
		change_variable = {
			name = bpm_seat_count_$KEY$
			add = var:bpmparl_leftover_seats
		}
	}
}

bpmparl_apply_party_range = {
	if = {
		limit = {
			var:bpm_seat_count_$KEY$ > 0
		}
		set_variable = {
			name = bpm_$COLOR$_parl_min
			value = var:bpm_allocated_seats
		}
		set_variable = {
			name = bpm_$COLOR$_parl_max
			value = {
				value = var:bpm_allocated_seats
				add = var:bpm_seat_count_$KEY$
				subtract = 1
			}
		}
		change_variable = {
			name = bpm_allocated_seats
			add = var:bpm_seat_count_$KEY$
		}
	}
}

bpmparl_clear_ranked_ig_slot = {
	set_variable = { name = bpm_$PREFIX$_share_$COLOR$ value = 0 }
	set_variable = { name = bpm_$PREFIX$_seat_count_$COLOR$ value = 0 }
	set_variable = { name = bpm_$PREFIX$_$COLOR$_parl_min value = 0 }
	set_variable = { name = bpm_$PREFIX$_$COLOR$_parl_max value = -1 }
	if = {
		limit = { has_variable = bpm_$PREFIX$_pointer_$COLOR$_ig }
		remove_variable = bpm_$PREFIX$_pointer_$COLOR$_ig
	}
}

bpmparl_cache_power_slot = {
	if = {
		limit = {
			any_interest_group = {
				ig_clout > 0
				count >= $REQUIRED$
			}
		}
		ordered_interest_group = {
			limit = {
				ig_clout > 0
			}
			order_by = ig_clout
			position = $POSITION$
			check_range_bounds = no
			owner = {
				set_variable = { name = bpm_power_pointer_$COLOR$_ig value = prev }
				set_variable = { name = bpm_power_share_$COLOR$ value = prev.ig_clout }
				set_variable = {
					name = bpm_power_seat_count_$COLOR$
					value = {
						value = prev.ig_clout
						divide = bpmparl_total_active_ig_clout
						multiply = bpmparl_number_of_seats
						floor = yes
					}
				}
				change_variable = {
					name = bpm_power_counted_share
					add = prev.ig_clout
				}
				change_variable = {
					name = bpm_power_total_counted_seats
					add = var:bpm_power_seat_count_$COLOR$
				}
				if = {
					limit = {
						var:bpm_power_seat_count_$COLOR$ > var:bpm_power_leading_seat_count
					}
					set_variable = { name = bpm_power_leading_seat_count value = var:bpm_power_seat_count_$COLOR$ }
					set_variable = { name = bpm_power_winning_slot value = flag:$COLOR$ }
				}
			}
		}
	}
}

						var:bpm_alignment_seat_count_$COLOR$ > var:bpm_alignment_leading_seat_count
					}
					set_variable = { name = bpm_alignment_leading_seat_count value = var:bpm_alignment_seat_count_$COLOR$ }
					set_variable = { name = bpm_alignment_winning_slot value = flag:$COLOR$ }
				}
			}
		}
	}
}

bpmparl_allocate_remaining_ranked_seats = {
	if = {
		limit = {
			has_variable = bpm_$PREFIX$_winning_slot
			var:bpm_$PREFIX$_leftover_seats > 0
			var:bpm_$PREFIX$_winning_slot = flag:$COLOR$
		}
		change_variable = {
			name = bpm_$PREFIX$_seat_count_$COLOR$
			add = var:bpm_$PREFIX$_leftover_seats
		}
	}
}

bpmparl_apply_ranked_ig_range = {
	if = {
		limit = {
			var:bpm_$PREFIX$_seat_count_$COLOR$ > 0
		}
		set_variable = {
			name = bpm_$PREFIX$_$COLOR$_parl_min
			value = var:bpm_$PREFIX$_allocated_seats
		}
		set_variable = {
			name = bpm_$PREFIX$_$COLOR$_parl_max
			value = {
				value = var:bpm_$PREFIX$_allocated_seats
				add = var:bpm_$PREFIX$_seat_count_$COLOR$
				subtract = 1
			}
		}
		change_variable = {
			name = bpm_$PREFIX$_allocated_seats
			add = var:bpm_$PREFIX$_seat_count_$COLOR$
		}
	}
}
