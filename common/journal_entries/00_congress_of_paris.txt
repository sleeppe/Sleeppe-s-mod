je_congress_of_paris = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_historical_content

	is_shown_in_lobby = {
		OR = {
			c:RUS ?= THIS
			c:TUR ?= THIS
		}
	}

	scripted_button = favour_russia_button
    scripted_button = favour_ottoman_button
    scripted_button = paris_congress_approve_button
    scripted_button = paris_congress_disapprove_button

	immediate = {
		c:RUS ?= {
			save_scope_as = russia_scope
		}
		c:TUR ?= {
			save_scope_as = ottoman_scope
		}

		if = {
			limit = {
				NOT = {
					exists = global_var:je_congress_of_paris_proposal_1
				}
			}
			set_global_variable = {
				name = je_congress_of_paris_proposal_1
				value = 0
			}
		}
		if = {
			limit = {
				NOT = {
					exists = global_var:je_congress_of_paris_proposal_2
				}
			}
			set_global_variable = {
				name = je_congress_of_paris_proposal_2
				value = 0
			}
		}
		if = {
			limit = {
				NOT = {
					exists = global_var:je_congress_of_paris_proposal_3
				}
			}
			set_global_variable = {
				name = je_congress_of_paris_proposal_3
				value = 0
			}
		}
		if = {
			limit = {
				NOT = {
					exists = global_var:je_congress_of_paris_proposal_4
				}
			}
			set_global_variable = {
				name = je_congress_of_paris_proposal_4
				value = 0
			}
		}
		if = {
			limit = {
				NOT = {
					exists = global_var:je_congress_of_paris_timer
				}
			}
			set_global_variable = {
				name = je_congress_of_paris_timer
				value = 6
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_congress_of_paris_tt_1
			has_global_variable = paris_treaty_has_concluded_global_var
		}
		custom_tooltip = {
			text = je_congress_of_paris_tt_2
			has_global_variable = paris_treaty_signed_rus_global_var
		}
		custom_tooltip = {
			text = je_congress_of_paris_tt_4
			has_global_variable = paris_treaty_signed_tur_global_var
		}
	}

	on_complete = {
		trigger_event = { id = treaty_of_paris_events.4 popup = yes }
	}

	fail = {
		OR = {
			custom_tooltip = {
				text = je_congress_of_paris_tt_3
				OR = {
					has_global_variable = paris_treaty_not_signed_tur_global_var
					has_global_variable = paris_treaty_not_signed_rus_global_var
				}
			}
			custom_tooltip = {
				text = je_congress_of_paris_tt_5
				has_war_with = c:ENG 
				has_war_with = c:FRA 
				has_war_with = c:TUR
				}
			}
		}
	

	on_fail = {
		trigger_event = { id = treaty_of_paris_events.5 popup = yes }
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = { # London Conference Timer: Only one country can actually modify this, or the congress will end after one month
					c:TUR ?= this
					global_var:je_congress_of_paris_timer >= 1
				}
				change_global_variable = {
					name = je_congress_of_paris_timer
					add = -1
				}
			}
			if = {
				limit = { # RUS/TUR Acceptance Timer: Only one country can actually modify this, or the congress will end after one month
					c:TUR ?= this
					exists = global_var:je_congress_of_paris_acceptance_timer
					global_var:je_congress_of_paris_acceptance_timer >= 1
				}
				change_global_variable = {
					name = je_congress_of_paris_acceptance_timer
					add = -1
				}
			}
			if = {
				limit = {
					c:TUR ?= this
					global_var:je_congress_of_paris_timer <= 0
				}
				if = { # If you have not chosen a proposal by now, a random one will be chosen for you. Must run in the c:TUR scope for sequence to work properly
					limit = {
						NOT = {
							has_variable = je_congress_of_paris_random_votes_var
						}
						any_country = {
							has_journal_entry = je_congress_of_paris
							NOR = {
								c:TUR ?= this
								c:RUS ?= this
							}
							NOR = {
								has_variable = favour_russia_button_var # proposal 1
								has_variable = favour_ottoman_button_var #proposal 2
								has_variable = partition_compromise_button_var # proposal 3
								has_variable = luxembourg_compromise_button_var # proposal 4
							}
						}
					}
				}
					set_variable = je_congress_of_paris_random_votes_var # this effect should only run once for performance reasons
					every_country = {
						limit = {
							has_journal_entry = je_congress_of_paris
							NOR = {
								c:TUR ?= this
								c:RUS ?= this
							}
							NOR = {
								has_variable = favour_russia_button_var # proposal 1
								has_variable = favour_ottoman_button_var #proposal 2
								has_variable = partition_compromise_button_var # proposal 3
								has_variable = luxembourg_compromise_button_var # proposal 4
							}
						}
						random_list = {
							1 = {
								set_variable = favour_russia_button_var
								change_global_variable = {
									name = je_congress_of_paris_proposal_1
									add = THIS.prestige
								}
								scope:russia_scope = {
									change_relations = { country = root value = 20 }
									create_diplomatic_catalyst = {
										type = catalyst_event_positive
										target = root
									}
								}
								scope:ottoman_scope = {
									change_relations = { country = root value = -20 }
									create_diplomatic_catalyst = {
										type = catalyst_event_negative
										target = root
									}
								}
							}
							1 = {
								set_variable = favour_ottoman_button_var
								change_global_variable = {
									name = je_congress_of_paris_proposal_2
									add = THIS.prestige
								}
								scope:ottoman_scope = {
									change_relations = { country = root value = 20 }
									create_diplomatic_catalyst = {
										type = catalyst_event_positive
										target = root
									}
								}
								scope:russia_scope = {
									change_relations = { country = root value = -20 }
									create_diplomatic_catalyst = {
										type = catalyst_event_negative
										target = root
									}
								}
							}
						} 
					}
				if = {
					limit = {
						NOR = {
							has_global_variable = favour_russia_global_var
							has_global_variable = favour_ottoman_global_var
						}
					}
					# Clear winners
					if = {
						limit = {
							global_var:je_congress_of_paris_proposal_1 > global_var:je_congress_of_paris_proposal_2
							global_var:je_congress_of_paris_proposal_1 > global_var:je_congress_of_paris_proposal_3
							global_var:je_congress_of_paris_proposal_1 > global_var:je_congress_of_paris_proposal_4
						}
						set_global_variable = favour_russia_global_var
					}
					else_if = {
						limit = {
							global_var:je_congress_of_paris_proposal_2 > global_var:je_congress_of_paris_proposal_1
							global_var:je_congress_of_paris_proposal_2 > global_var:je_congress_of_paris_proposal_3
							global_var:je_congress_of_paris_proposal_2 > global_var:je_congress_of_paris_proposal_4
						}
						set_global_variable = favour_ottoman_global_var
					}

					else_if = { # Tiebreakers
						limit = {
							global_var:je_congress_of_paris_proposal_1 = global_var:je_congress_of_paris_proposal_2
							global_var:je_congress_of_paris_proposal_1 > global_var:je_congress_of_paris_proposal_3
							global_var:je_congress_of_paris_proposal_1 > global_var:je_congress_of_paris_proposal_4
						}
						ordered_country = {
							limit = {
							    has_journal_entry = je_congress_of_paris
							    OR = {
							    	has_variable = favour_russia_button_var # proposal 1
							    	has_variable = favour_ottoman_button_var #proposal 2
							    }
							}
						    order_by = country_rank
							position = 0
							check_range_bounds = no
						    if = {
						        limit = {
						            has_variable = favour_russia_button_var
						        }
						        set_global_variable = favour_russia_global_var
						    }
						    else_if = {
						        limit = {
						            has_variable = favour_ottoman_button_var
						        }
						        set_global_variable = favour_ottoman_global_var
						    }
						}
					}
				 	else = {
						ordered_country = {
							limit = {
							    has_journal_entry = je_congress_of_paris
							}
						    order_by = country_rank
							position = 0
							check_range_bounds = no
						    if = {
						        limit = {
						            has_variable = favour_russia_button_var
						        }
						        set_global_variable = favour_russia_global_var
						    }
						    if = {
						        limit = {
						            has_variable = favour_ottoman_button_var
						        }
						        set_global_variable = favour_ottoman_global_var
						    }
						}
					}
					set_global_variable = paris_treaty_has_concluded_global_var
				}
			}
			if = {
				limit = {
					c:TUR ?= this
					has_global_variable = paris_treaty_has_concluded_global_var
					NOT = {
						exists = global_var:je_congress_of_paris_acceptance_timer
					}
				}
				set_global_variable = {
					name = je_congress_of_paris_acceptance_timer
					value = 6
				}
			}
			if = {
				limit = {
					exists = global_var:je_congress_of_paris_acceptance_timer
					OR = {
						c:TUR ?= this
						c:RUS ?= this
					}
					NOT = {
						has_variable = decided_on_paris_treaty_var
					}
					global_var:je_congress_of_paris_acceptance_timer <= 0
				}
				set_variable = decided_on_paris_treaty_var
				if = {
					limit = {
						c:TUR = this
					}
					set_global_variable = paris_treaty_signed_tur_global_var
				}
				if = {
					limit = {
						c:RUS = this
					}
					set_global_variable = paris_treaty_signed_rus_global_var
				}
			}
		}
	}
	on_yearly_pulse = {
	}

	weight = 100

	should_be_pinned_by_default = yes


}
