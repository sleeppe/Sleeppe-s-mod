##Modded Decisions for the Journal Tab



polish_constitution = {
	is_shown = {
		c:POL = ROOT
		OR = {
			NOT = { exists = global_var:polish_constitution_global }
		}
		c:POL = {
			is_subject_of= c:RUS
		}
	}

	possible = {
		has_technology_researched = nationalism
	}

	when_taken = {
		if = {
			limit = {
				NOT = { has_global_variable = polish_constitution_global }
			}
			set_global_variable = {
				name = polish_constitution_global
				value = 0
			}
		}
		add_modifier = {
			name = modifier_signing_constitution
			days = -1 # permanent until removed
		}
		add_journal_entry = { type = je_polish_constitution }
	}
	ai_chance = {
		value = 10
	}
}
