budget_mil_increase_button = {
	name = "budget_mil_increase"
	desc = "Increase Military Budget"
	
	effect = {
		change_variable = {
			name = budget_mil
			add = 1
		}
		change_variable = {
			name = budget_total
			add = 1
		}
	}
}

budget_mil_decrease_button = {
	name = "budget_mil_decrease"
	desc = "Decrease Military Budget"
	
	effect = {
		change_variable = {
			name = budget_mil
			add = -1
		}
		change_variable = {
			name = budget_total
			add = -1
		}
	}
}
