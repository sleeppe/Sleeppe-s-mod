law_mercantilism = {
	group = lawgroup_trade_policy

	icon = "gfx/interface/icons/law_icons/mercantilism.dds"

	progressiveness = 0

	unlocking_technologies = {
		international_trade
	}

	is_visible = { 
		NOT = {
		c:USA ?= this
		}
	}

	modifier = {
		state_tariff_import_add = 0.50
		state_tariff_export_add = 0.20
		state_subvention_export_add = 0.25
		state_subvention_import_add = 0.20
		state_import_advantage_mult = -0.25
		state_export_advantage_mult = 0.25
	}

	# Don't change back to mercantilism once we have protectionism or free trade
	ai_will_do = {
		NOR = {
			has_law_or_variant = law_type:law_protectionism
			has_law_or_variant = law_type:law_free_trade
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}

		if = {
			limit = {
				has_journal_entry = je_meiji_diplomacy
				has_law_or_variant = law_type:law_isolationism
			}
			add = 1000
		}
	}
}

law_free_trade = {
	group = lawgroup_trade_policy

	icon = "gfx/interface/icons/law_icons/free_trade.dds"

	progressiveness = 50

	disallowing_laws = {
		law_command_economy
	}

	unlocking_technologies = {
		international_trade
	}

	on_enact = {
		if = {
			limit = { is_banning_goods = g:opium }
			remove_banned_goods = g:opium
		}
		if = {
			limit = { is_banning_goods = g:liquor }
			remove_banned_goods = g:liquor
		}
	}

	on_activate = {
		if = { # removes opium ban authority modifier if forced to open market via diplomacy
			limit = {
				AND = {
					c:CHI ?= this
					has_modifier = opium_ban_authority
				}
			}
			remove_modifier = opium_ban_authority
		}
	}
    is_visible = {
		NOT = {
			c:USA ?= this
		}
	}
	modifier = {
		state_subvention_export_add = 0.50
		state_subvention_import_add = 0.50
		state_trade_advantage_mult = 0.50
		country_leverage_resistance_mult = -0.25
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}

		if = {
			limit = {
				has_journal_entry = je_meiji_diplomacy
				has_law_or_variant = law_type:law_isolationism
			}
			add = 1000
		}
	}
}

law_isolationism = {
	group = lawgroup_trade_policy

	icon = "gfx/interface/icons/law_icons/isolationism.dds"

	progressiveness = -50

	disallowing_laws = {
		law_laissez_faire
	}

	is_visible = {
		NOT = { c:USA ?= this }
		NAND = {
			any_primary_culture = {
				has_discrimination_trait = heritage_han
			}
			any_scope_state = {
				state_region = s:STATE_GUANGDONG
				is_largest_state_in_region = yes
			}
		}

	}

	modifier = {
		country_authority_mult = 0.5
		state_tax_capacity_mult = 0.25
		country_tech_spread_mult = -0.15

		country_support_separatism_resistance_mult = 0.5

		country_max_declared_interests_mult = -0.5

		country_disallow_trade_bool = yes

	ai_will_do = {
		OR = {

					has_ideology = ideology:ideology_traditionalist
