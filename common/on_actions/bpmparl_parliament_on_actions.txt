on_game_started_after_lobby = {
	on_actions = {
		bpmparl_game_started_on_action
	}
}

on_monthly_pulse_country = {
	on_actions = {
		bpmparl_parliament_update_on_action
	}
}

on_election_campaign_end = {
	on_actions = {
		bpmparl_parliament_update_on_action
	}
}

on_law_activated = {
	on_actions = {
		bpmparl_law_activated_on_action
	}
}

on_monthly_pulse_country_elections = {
	on_actions = {
		bpmparl_parliament_update_on_action
	}
}

bpmparl_game_started_on_action = {
	effect = {
		
			bpmparl_update_parliament_seats = yes
	}
}


bpmparl_law_activated_on_action = {
	effect = {
		owner = {
			if = {
				bpmparl_update_parliament_seats = yes
			}
		}
	}
}

bpmparl_parliament_update_on_action = {
	effect = {
		if = {
			bpmparl_update_parliament_seats = yes
		}
	}
}
